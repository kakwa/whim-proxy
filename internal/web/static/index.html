whim-proxy

Reverse-tunnel webhooks to your laptop

version: {{.Version}}

Quick start

# 1. Download client (see below)

# 2. Generate a channel UUID (can be reused across sessions)
CHANNEL=$(./whim-client --gen-uuid)
echo "$CHANNEL"

# 3. Start the client on your laptop
TARGET=http://localhost:8080/webhook
./whim-client --server ws://{{.Host}} --channel "$CHANNEL" --target "$TARGET"

# 4. Configure your webhook sender to POST to:
echo "http://{{.Host}}/hook/$CHANNEL"

# 5. Check received events:
./whim-client --server ws://{{.Host}} --channel "$CHANNEL" --logs
